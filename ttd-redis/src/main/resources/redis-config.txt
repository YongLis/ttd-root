spring.redis.cluster.nodes=localhost:6379
spring.redis.database=4
# 连接超时时间（毫秒）
spring.redis.timeout=3000ms
# 最大活跃连接数
spring.redis.lettuce.pool.max-active=32
# 最大空闲连接数
spring.redis.lettuce.pool.max-idle=32
# 最小空闲连接数
spring.redis.lettuce.pool.min-idle=8
# 连接池最大阻塞等待时间
spring.redis.lettuce.pool.max-wait=1000ms
# 缓存类型：指定为Redis
spring.redis.cache.type=redis

# 缓存过期时间
spring.redis.cache.redis.time-to-live=3600000ms
# 是否允许缓存空值（防止缓存穿透）
spring.redis.cache.redis.cache-null-values=true
# 缓存键前缀
spring.redis.cache.redis.key-prefix="ttd:"
# 是否使用键前缀
spring.redis.cache.redis.use-key-prefix=true
